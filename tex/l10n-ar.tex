\startsetups page:cover

    \startalignment[flushleft]
      \framed[align=flushleft,width=.5\textwidth,frame=off]
        {\definedfont[Serif sa 5]\getvariable{l10n-manual}{title}}
      \blank[4*big]
      \bfc
      \getvariable{l10n-manual}{subtitle}
      \startmode[screen]
        \blank[4*big]
        \getvariable{l10n-manual}{author}
        \blank
        {\tfx ترجمة:}\par %XXX: make localisable
        \startluacode
            local t = string.split("\getvariable{l10n-manual}{translator}", ",")
            for _,v in next, t do % without the % newline gets gobbled!
                context(v)
                context.par()
            end
        \stopluacode
      \stopmode
    \stopalignment
  \stopstandardmakeup
\stopsetups
